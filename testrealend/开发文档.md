## 二、Python基础知识

### 2.1 装饰器(Decorator)

装饰器是Python的一个重要特性，用于在不修改函数代码的情况下扩展函数功能。

#### 2.1.1 基本原理

```python
# 装饰器本质是一个接受函数作为参数的函数
def my_decorator(func):
    def wrapper(*args, **kwargs):
        print("函数执行前")
        result = func(*args, **kwargs)  # 执行原函数
        print("函数执行后")
        return result
    return wrapper

# 使用装饰器
@my_decorator
def say_hello(name):
    print(f"Hello, {name}!")

# 等价于: say_hello = my_decorator(say_hello)
say_hello("World")
# 输出:
# 函数执行前
# Hello, World!
# 函数执行后
```

#### 2.1.2 带参数的装饰器

```python
def repeat(times):
    """重复执行装饰器"""
    def decorator(func):
        def wrapper(*args, **kwargs):
            for _ in range(times):
                result = func(*args, **kwargs)
            return result
        return wrapper
    return decorator

@repeat(times=3)
def greet(name):
    print(f"Hi, {name}!")

greet("Alice")  # 打印3次
```

#### 2.1.3 Flask中的装饰器应用

```python

from functools import wraps

app = Flask(__name__)

# Flask路由装饰器
@app.route('/api/hello')

def admin_required(f):
    @wraps(f)  # 保留原函数的元信息

        # 检查是否是管理员

            return {'error': '需要管理员权限'}, 403

@app.route('/admin/dashboard')
@admin_required

    return {'message': '管理员仪表板'}
```

### 2.2 类与对象

#### 2.2.1 类的基本结构

```python
class User:
    # 类属性（所有实例共享）
    role = 'user'
    
    # 构造方法
    def __init__(self, username, email):
        # 实例属性
        self.username = username
        self.email = email
    
    # 实例方法
    def get_info(self):
        return f"用户: {self.username}, 邮箱: {self.email}"
    
    # 类方法
    @classmethod
    def create_admin(cls, username):
        admin = cls(username, f"{username}@admin.com")
        admin.role = 'admin'
        return admin
    
    # 静态方法
    @staticmethod
    def validate_email(email):
        return '@' in email

# 使用
```

```python
# 生成器：惰性计算，节省内存
def generate_numbers(n):
    for i in range(n):
        yield i  # 每次只生成一个值

# 使用生成器
for num in generate_numbers(1000000):
    if num > 10:
        break

# 在数据库查询中的应用：分批处理大数据
def batch_query(query, batch_size=1000):
    offset = 0
    while True:
        batch = query.offset(offset).limit(batch_size).all()
        if not batch:
            break
        yield batch
        offset += batch_size
```

```python
        """转换为字典"""
        return {

            'username': self.username,
            'email': self.email,
            'role': self.role,
            'created_at': self.created_at.isoformat() if self.created_at else None
```

#### 5.2.2 密码哈希原理

```python
# werkzeug使用pbkdf2或scrypt算法

# generate_password_hash内部流程：
# 1. 生成随机盐(salt)
# 2. 使用scrypt/pbkdf2进行哈希
# 3. 格式: algorithm$iterations$salt$hash
# 示例: "scrypt:32768:8:1$salt$hash"

# 为什么不能直接比较密码？
# 1. 每次生成的哈希不同（因为盐不同）
# 2. 必须使用check_password_hash验证

# 本项目的密码匹配函数
def _password_matches(stored_password, input_password):
    if not stored_password or not input_password:
        return False
    try:
        # 尝试哈希验证
        if check_password_hash(stored_password, input_password):
            return True
    except Exception:
        pass
    # 兼容旧数据：如果不是哈希格式，则明文比较
    if not stored_password.startswith(('pbkdf2:', 'sha$', '$2b$', 'scrypt:')):
        return stored_password == input_password
    return False
```

```javascript
// stores/userStore.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useUserStore = defineStore('user', () => {
  // State
  const token = ref('')
  const currentUser = ref(null)
  
  // Getters (计算属性)
  const isAuthenticated = computed(() => !!token.value)
  const userName = computed(() => currentUser.value?.name || '游客')
  
  // Actions (方法)
  const setUserInfo = (user) => {
    currentUser.value = user
    if (user) {
      token.value = user.token || ''
      sessionStorage.setItem('token', token.value)
    }
  }
  
  const clearUserInfo = () => {
    token.value = ''
    currentUser.value = null
    sessionStorage.clear()
  }
  
  const initializeStore = () => {
    const storedToken = sessionStorage.getItem('token')
    if (storedToken) {
      token.value = storedToken
      // 可选：验证token有效性
    }
  }
  
  return {
    token,
    currentUser,
    isAuthenticated,
    userName,
    setUserInfo,
    clearUserInfo,
    initializeStore
  }
})
```

```javascript
    }
    return config
  },
  (error) => {

  }
)

  (response) => {

  },

          break

          ElMessage.error('请求的资源不存在')
          break
        case 500:
          ElMessage.error('服务器错误')
          break

          ElMessage.error(error.response.data?.msg || '请求失败')
      }
    } else if (error.request) {
      ElMessage.error('网络错误，请检查网络连接')
    } else {
```

```python
# 审批状态流转
# pending -> approved/rejected

class ApprovalResource(Resource):
    @jwt_required()
    def post(self, application_id):
        """审批申请"""
        identity = get_jwt_identity()
        approver_number = identity.get('number')
        
        application = Application.query.get_or_404(application_id)
        
        # 检查是否可以审批
        if application.adm1_statu is not None and application.adm1_user_number != approver_number:
            # 已经一审过了，检查二审权限
            if identity.get('role') != 'admin':
                return {'status': False, 'msg': '无审批权限'}, 403
        
        data = request.get_json()
        approved = data.get('approved', False)
        comment = data.get('comment', '')
        
        if application.adm1_statu is None:
            # 一审
            application.adm1_statu = approved
            application.adm1_user_number = approver_number
            application.adm1_time = datetime.utcnow()
            application.adm1_comment = comment
            
            if not approved:
                # 一审拒绝，流程结束
                application.status = 'rejected'
        else:
            # 二审
            application.adm2_statu = approved
            application.adm2_user_number = approver_number
            application.adm2_time = datetime.utcnow()
            application.adm2_comment = comment
            
            if approved:
                application.status = 'approved'
                # 触发水印处理
                process_watermark(application)
            else:
                application.status = 'rejected'
        
        db.session.commit()
        
        return {
            'status': True,
            'msg': '审批成功',
            'data': application.to_dict()
        }
```

### 8.3 文件上传下载

```python
import os
from werkzeug.utils import secure_filename

ALLOWED_EXTENSIONS = {'shp', 'shx', 'dbf', 'tif', 'tiff'}

def allowed_file(filename):
    return '.' in filename and filename.rsplit('.', 1)[1].lower() in ALLOWED_EXTENSIONS

class FileUploadResource(Resource):
    @jwt_required()
    def post(self):
        """上传文件"""
        if 'file' not in request.files:
            return {'status': False, 'msg': '没有文件'}, 400
        
        file = request.files['file']
        if file.filename == '':
            return {'status': False, 'msg': '没有选择文件'}, 400
        
        if not allowed_file(file.filename):
            return {'status': False, 'msg': '不支持的文件类型'}, 400
        
        # 安全文件名
        filename = secure_filename(file.filename)
        # 添加时间戳避免冲突
        filename = f"{datetime.now().strftime('%Y%m%d%H%M%S')}_{filename}"
        
        # 保存路径
        upload_dir = app.config['UPLOAD_FOLDER']
        os.makedirs(upload_dir, exist_ok=True)
        filepath = os.path.join(upload_dir, filename)
        
        file.save(filepath)
        
        return {
            'status': True,
            'data': {
                'filename': filename,
                'filepath': filepath
            }
        }


class FileDownloadResource(Resource):
    @jwt_required()
    def get(self, filename):
        """下载文件"""
        filepath = os.path.join(app.config['UPLOAD_FOLDER'], filename)
        
        if not os.path.exists(filepath):
            return {'status': False, 'msg': '文件不存在'}, 404
        
        # 记录下载日志
        download_record = DownloadRecord(
            user_number=get_jwt_identity().get('number'),
            filename=filename,
            download_time=datetime.utcnow()
        )
        db.session.add(download_record)
        db.session.commit()
        
        return send_file(filepath, as_attachment=True)
```
